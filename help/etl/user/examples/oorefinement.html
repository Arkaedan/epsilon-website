OO Refinement Transformation

auto rule StructuralFeature2GetterAndSetter
  transform s : Source!StructuralFeature
  to t : Target!StructuralFeature {
  
  t.visibility := Target!VisibilityEnum#private;
  
  var getter : new Target!Operation;
  getter.name := 'get' + s.name.firstToUpperCase();
  getter.owner := t.owner;
  getter.type := t.type;
  
  var setter : new Target!Operation;
  setter.name := 'set' + s.name.firstToUpperCase();
  setter.owner := t.owner;
  
  var inputParameter : new Target!Parameter;
  inputParameter.name := t.name;
  inputParameter.type := t.type;
  setter.parameters := Sequence{inputParameter};
  
}
